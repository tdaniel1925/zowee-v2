ZoweeHeader

ZoweeHeader Props

TypeScript
Prop Type Default Description
variant 'landing' | 'dashboard' | 'rep' 'landing' Controls the header layout, badge, and navigation items rendered
planName 'Pro' | 'Starter' | 'Trial' undefined Dashboard variant only — renders the plan badge next to the logo
repCode string undefined Rep variant only — displays as REP-{repCode} badge with animated dot
sticky boolean true Enables sticky positioning with blur backdrop on scroll
navLinks NavLink[] auto Override default nav links. Each item: {label, href, active?}
onBillingClick () => void undefined Dashboard variant — callback for the Billing button click
user UserObject undefined Authenticated user — renders avatar + name in dashboard/rep variants
className string undefined Additional CSS classes merged onto the header root element

Code Snippet

import { ZoweeHeader } from '@zowee/components';

// Landing / marketing pages
// Shows: logo + nav links + Get Started CTA
<ZoweeHeader
  variant="landing"
  sticky
  navLinks={[
    { label: 'Features', href: '/features' },
    { label: 'Pricing',  href: '/pricing'  },
    { label: 'About',    href: '/about'    },
    { label: 'Blog',     href: '/blog'     },
  ]}
/>
import { ZoweeHeader } from '@zowee/components';

// Subscriber dashboard
// Shows: logo + plan badge + nav + billing button + avatar
<ZoweeHeader
  variant="dashboard"
  sticky
  planName="Pro"
  user={{ name: 'Sarah K.', avatar: user.avatarUrl }}
  navLinks={[
    { label: 'Dashboard', href: '/dashboard', active: true },
    { label: 'Messages',  href: '/messages'  },
    { label: 'Settings',  href: '/settings'  },
  ]}
  onBillingClick={() => router.push('/account/billing')}
/>
import { ZoweeHeader } from '@zowee/components';

// Rep Portal — authorized reps only
// Shows: logo + "Rep Portal" label + nav + rep code badge
<ZoweeHeader
  variant="rep"
  sticky
  repCode="ZOWEE42"
  user={{ name: 'Marcus T.', avatar: rep.avatarUrl }}
  navLinks={[
    { label: 'Overview',     href: '/rep-portal',              active: true },
    { label: 'Subscribers',  href: '/rep-portal/subscribers'  },
    { label: 'Commissions',  href: '/rep-portal/commissions'  },
  ]}
/>
