Variant Guide

primary

High-emphasis actions. CTAs, form submits, primary flows.

secondary

Medium-emphasis. Paired with primary, save/draft actions.

ghost

Low-emphasis. Cancel, dismiss, tertiary actions.

Component Props

TypeScript
Prop Type Default Description
variant 'primary' | 'secondary' | 'ghost' 'primary' Visual style of the button
size 'sm' | 'md' | 'lg' 'md' Controls padding and font size
loading boolean false Shows spinner and disables interaction
fullWidth boolean false Stretches button to 100% container width
href string undefined Renders as <a> tag when provided
disabled boolean false Disables button, reduces opacity to 0.38
onClick function undefined Click handler. Not called when disabled or loading
children ReactNode required Button label and optional icon elements

Code Snippet

import { ZoweeButton } from '@zowee/components';

// Primary — default high-emphasis CTA
<ZoweeButton variant="primary" size="md">
  Submit
</ZoweeButton>

// Secondary — medium-emphasis paired action
<ZoweeButton variant="secondary" size="md">
  Save Draft
</ZoweeButton>

// Ghost — low-emphasis dismiss/cancel
<ZoweeButton variant="ghost" size="md">
  Cancel
</ZoweeButton>

// Size variants
<ZoweeButton variant="primary" size="sm">Small</ZoweeButton>
<ZoweeButton variant="primary" size="lg">Large</ZoweeButton>
import { ZoweeButton } from '@zowee/components';
import { useState } from 'react';

export default function SubmitForm() {
  const [isLoading, setIsLoading] = useState(false);

  const handleSubmit = async () => {
    setIsLoading(true);
    await submitData();
    setIsLoading(false);
  };

  return (
    <ZoweeButton
      variant="primary"
      size="md"
      loading={isLoading}
      onClick={handleSubmit}
    >
      {isLoading ? 'Submitting...' : 'Submit'}
    </ZoweeButton>
  );
}
import { ZoweeButton } from '@zowee/components';

// When href is provided, renders as <a> tag
// Supports all standard anchor attributes

<ZoweeButton
  variant="primary"
  size="md"
  href="/dashboard"
>
  Go to Dashboard
</ZoweeButton>

// External link with target
<ZoweeButton
  variant="secondary"
  size="sm"
  href="https://docs.zowee.com"
  target="_blank"
  rel="noopener noreferrer"
>
  View Docs ↗
</ZoweeButton>
