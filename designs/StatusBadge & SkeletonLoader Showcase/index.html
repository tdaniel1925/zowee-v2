SkeletonLoader Props

TypeScript
Prop Type Default Description
variant 'card' | 'text' | 'circle' 'card' Layout pattern — card mirrors ZoweeCard, text for paragraphs, circle for avatars
animation 'pulse' | 'shimmer' | 'none' 'pulse' Pulse fades opacity; shimmer sweeps a highlight across the element
rows number 3 Number of repeated skeleton rows (text and circle variants)
stagger boolean true Offsets animation-delay per row for a cascading effect
className string undefined Additional CSS classes merged onto the skeleton root element

Code Snippet

import { StatusBadge } from '@zowee/components';

// All four status variants
<StatusBadge status="active" />
// → green dot + "Active" label

<StatusBadge status="trial" />
// → cyan dot + "Trial" label

<StatusBadge status="past_due" />
// → amber dot + "Past Due" label

<StatusBadge status="canceled" />
// → static red dot + "Canceled" label

// Custom label override
<StatusBadge
  status="active"
  label="Subscribed"
  size="sm"
/>
import { SkeletonLoader } from '@zowee/components';

// Card skeleton — mirrors ZoweeCard layout
<SkeletonLoader variant="card" animation="shimmer" />

// Text skeleton — 5 lines with stagger
<SkeletonLoader
  variant="text"
  rows={5}
  animation="pulse"
  stagger
/>

// Circle skeleton — avatar + name rows
<SkeletonLoader
  variant="circle"
  rows={4}
  animation="pulse"
  stagger
/>
import { StatusBadge, SkeletonLoader } from '@zowee/components';

// Async user list with loading state
export default function UserList({ users, isLoading }) {
  if (isLoading) {
    return <SkeletonLoader variant="circle" rows={4} stagger />;
  }

  return (
    <ul>
      {users.map(user => (
        <li key={user.id} className="user-row">
          <img src={user.avatar} alt={user.name} />
          <div>
            <p>{user.name}</p>
            <p>{user.email}</p>
          </div>
          <StatusBadge
            status={user.subscriptionStatus}
            size="sm"
          />
        </li>
      ))}
    </ul>
  );
}
